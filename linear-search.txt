//Linear Search:
#include <iostream> 
using namespace std; 
  
 
int search(int arr[], int n, int x) 
{ 
    int i; 
    for (i = 0; i < n; i++) 
        if (arr[i] == x) 
            return i; 
    return -1; 
} 
  
// Driver code 
int main() 
{ 
    int n,i,x;
    cin>>n;
    int arr[n];
    for(i=0;i<n;i++){
        cin>>arr[i];
    }
    
    cout<<"Enter element to be searched:\t";
    cin>>x;
  
    int index = search(arr, n, x); 
    if (index == -1) 
        cout << "Element is not present in the array"; 
    else
        cout << "Element found at position " << index; 
  
    return 0; 
}
